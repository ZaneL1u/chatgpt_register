```python
"""拟人化邮箱前缀生成器。"""

from __future__ import annotations

import random
import threading


class HumanizedPrefixGenerator:
    """生成真实人名格式的邮箱前缀，线程安全且保证唯一性。"""

    _used: set[str] = set()
    _lock = threading.Lock()

    def generate(self) -> str:
        """生成一个唯一的拟人化邮箱前缀。"""
        with self._lock:
            retries = 0
            while retries < 1000:
                prefix = self._make_prefix()
                if prefix not in self._used:
                    self._used.add(prefix)
                    return prefix
                retries += 1
            # 超过重试上限，fallback 到随机字符串
            import string
            fallback = "".join(
                random.choice(string.ascii_lowercase + string.digits)
                for _ in range(random.randint(8, 13))
            )
            self._used.add(fallback)
            return fallback

    def _make_prefix(self) -> str:
        first, last = self._get_names()
        fmt = random.randint(0, 3)
        year = random.randint(1980, 2006)
        if fmt == 0:
            return f"{first}.{last}"
        elif fmt == 1:
            return f"{first}_{year}"
        elif fmt == 2:
            return f"{first}{year % 100:02d}"
        else:
            return f"{first[0]}.{last}"

    def _get_names(self) -> tuple[str, str]:
        try:
            from faker import Faker
            fake = Faker("en_US")
            return fake.first_name().lower(), fake.last_name().lower()
        except ImportError:
            pass
        try:
            import names
            return names.get_first_name().lower(), names.get_last_name().lower()
        except ImportError:
            raise RuntimeError(
                "faker 和 names 均未安装，无法生成拟人化邮箱前缀"
            )

    @classmethod
    def reset(cls) -> None:
        """清空已用前缀集合（仅供测试使用）。"""
        with cls._lock:
            cls._used.clear()
```

**关键设计点：**
- `_used` 和 `_lock` 为类变量，跨所有实例共享，确保全局唯一性
- Faker 实例在 `_get_names()` 中实时创建（按 CONTEXT.md 锁定决策），不缓存
- `_lock` 保护整个 generate 过程，确保 check-then-add 原子性
- 重试上限 1000 次，超限 fallback 到随机字符串（防止极端场景死循环）
- `reset()` 类方法仅供测试使用
